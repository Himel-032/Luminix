case_item
    : CASE expression COLON
        {
            SwitchFrame *f = switch_cur();
            if (f == NULL) {
                yyerror("case outside switch");
                execute_flag = 0;
            } else if (f->done) {
                /* break already hit — skip everything */
                execute_flag = 0;
            } else if (f->matched) {
                /* fall-through: previous case ran without break */
                execute_flag = 1;
            } else if (fabs(f->value - $2) < SWITCH_EPSILON) {
                /* first matching case */
                f->matched   = 1;
                execute_flag = 1;
            } else {
                /* not matched yet */
                execute_flag = 0;
            }
        }
      statement_list
        {
            SwitchFrame *f = switch_cur();
            if (f && break_flag && execute_flag) {
                f->done      = 1;
                execute_flag = 0;
                break_flag   = 0;
            }
        }
