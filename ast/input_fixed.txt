StartExam(){

    Number a = 10;
    Number b = 20;
    Number c;

    TrueFalse flag = False;

    ## arithmetic assignment
    c = a + b;

    Show(c);

    Show("Value of x:");
    Show(x);

    Show("Value of y:");
    Show(y);

    Show(ch);

    Show(flag);

    Show(matrix[1][2]);
