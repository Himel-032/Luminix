StartExam(){
    ## Test 1: Triple nested loops
    Show("=== Test 1: Triple Nested Loops ===");
    Repeat(Number i = 0; i < 2; i = i + 1){
        Show("Level 1 i =");
        Show(i);
        Repeat(Number j = 0; j < 2; j = j + 1){
            Show("  Level 2 j =");
            Show(j);
            Repeat(Number k = 0; k < 2; k = k + 1){
                Show("    Level 3 k =");
                Show(k);
            }
        }
    }
    
    ## Test 2: Mix of loop types
    Show("=== Test 2: Mix of While, For, Do-While ===");
    Number a = 0;
    WhileStudy(a < 2){
        Show("While iteration");
        Repeat(Number b = 0; b < 2; b = b + 1){
            Show("  For in While");
        }
        Number c = 0;
        DoOnce{
            Show("  Do-While in While");
            c = c + 1;
        }
        WhileStudy(c < 1);
        a = a + 1;
    }
    
    ## Test 3: Break from deeply nested loop
    Show("=== Test 3: Break from Inner Nested Loop ===");
    Repeat(Number x = 0; x < 3; x = x + 1){
        Show("Outer x =");
        Show(x);
        Repeat(Number y = 0; y < 5; y = y + 1){
            Check(y == 2){
                Show("  Breaking inner loop at y=2");
                Stop;
            }
            Show("  Inner y =");
            Show(y);
        }
        Show("  Outer continues");
    }
    
    ## Test 4: Continue in deeply nested
    Show("=== Test 4: Continue in Nested Loops ===");
    Repeat(Number m = 0; m < 2; m = m + 1){
        Repeat(Number n = 0; n < 4; n = n + 1){
            Check(n == 2){
                Show("    Skipping n=2 in m =");
                Show(m);
                Skip;
            }
            Show("  m,n =");
            Show(m);
            Show(n);
        }
    }
    
    ## Test 5: Empty for loop (infinite loop with manual break)
    Show("=== Test 5: For Loop Variations ===");
    Number counter = 0;
    Repeat(Number idx = 0; idx < 5; idx = idx + 1){
        counter = counter + 1;
    }
    Show("Counter =");
    Show(counter);
    
    ## Test 6: While with complex conditions
    Show("=== Test 6: While with Complex Logic ===");
    Number val = 10;
    WhileStudy(val > 5){
        Check(val == 8){
            Show("Value is 8, skipping");
            val = val - 1;
            Skip;
        }
        Check(val == 6){
            Show("Value is 6, breaking");
            Stop;
        }
        Show("Value =");
        Show(val);
        val = val - 1;
    }
    
    Show("=== All Advanced Tests Completed ===");
    Return 0;
